Includes:

* county irrigated area - Census
* county total cropland area - Census

Notes:

# Irrigated Area
Run 1x

census > economics > farms and land and assets > ag land > area >

Run for all states


``` r
#statesWanted <- c('IN','IL','IA','OH','MI','MO','MN','WI','MI','NE','SD','ND')

# param list without state
paramsNoState <- list("source_desc" = "CENSUS",
                 "agg_level_desc" = 'COUNTY',
                 "year__GE" = 1997,
                 "short_desc" = 'AG LAND, IRRIGATED - ACRES',
                 "domain_desc" = 'TOTAL')


# do all states together
nassdf <- nassqs(paramsNoState)


# # split up by state to avoid query limits
# nasslist <- list()
# for(state in statesWanted){
#   params <- c(list("state_alpha" = state), paramsNoState)
#   # retrieve county data by state
#   nasslist[[state]] <- nassqs(params)
# }

# # convert list of data frames to 1 giant dataframe
# nass.df <- do.call("rbind",nasslist)

# tidy up a bit
nassIrr <- nassdf %>%
  mutate(fips5 = paste0(state_fips_code, county_ansi)) %>%
  filter(!is.na(Value)) %>%    # remove no data
  mutate(irrArea_ha = Value / 2.471) %>%  # acres to ha
  dplyr::rename(irrArea_acres = Value) %>%
  dplyr::select(c(fips5, county_name, state_alpha, state_name, year, short_desc,
                  domain_desc, irrArea_acres, irrArea_ha)) %>%
  arrange(fips5, year) %>%
  as_tibble()

# export
write_csv(nassIrr, paste0(nassDir, '/', nassFileName_raw_irrigated))
```
